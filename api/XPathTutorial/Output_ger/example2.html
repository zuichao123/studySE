Falls der Pfad mit // beginnt, werden alle Elemente ausgewählt, welche die folgenden Kriterien erfüllen.
 
//BBB
Wähle alle BBB Elemente aus

     <AAA>
          <BBB/>
          <CCC/>
          <BBB/>
          <DDD>
               <BBB/>
          </DDD>
          <CCC>
               <DDD>
                    <BBB/>
                    <BBB/>
               </DDD>
          </CCC>
     </AAA>
