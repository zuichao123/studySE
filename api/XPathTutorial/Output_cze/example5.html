Atributy jsou specifikovány pomocí prefixu "@".
 
//@id
Vyber všechny atributy @id

     <AAA>
          <BBB id = "b1"/>
          <BBB id = "b2"/>
          <BBB name = "bbb"/>
          <BBB/>
     </AAA>
