练习实例1：
<!ELEMENT TVSCHEDULE (CHANNEL + )>					//元素TVSCHEDULE的下级元素CHANNEL一次或多次
<!ELEMENT CHANNEL (BANNER,DAY + )>					//元素CHANNEL的下级元素BANNER必须有；DAY一次或多次
<!ELEMENT BANNER (#PCDATA)>							//随便写
<!ELEMENT DAY (DATE,(HOLIDAY|PROGRAMSLOT + )+)>		//元素CHANNEL的下级元素DATE必须有；括号里任选其一，HOLIDAY有一次；PROGRAMSLOT一次或多次
<!ELEMENT HOLIDAY (#PCDATA)>						//元素DAY的下级元素选了这个选不了PROGRAMSLOT
<!ELEMENT DATE (#PCDATA)>							//元素DAY的下级元素DAY里边的必须有
<!ELEMENT PROGRAMSLOT (TIME,TITLE,DESCRIPTION?)>	//元素DAT的下级元素TIME必须有、TITLE必须有、DESCRIPTION可有可无
<!ELEMENT TIME (#PCDATA)>							//元素PROGRAMSLOT的下级元素time
<!ELEMENT TITLE (#PCDATA)>							//元素PROGRAMSLOT的下级元素title
<!ELEMENT DESCRIPTION (#PCDATA)>					//元素PROGRAMSLOT的下级元素description
<!ATTLIST TVSCHEDULE NAME CDATA #REQUIRED>			//元素TVSCHEDULE的属性“NAME”，是文本数据
<!ATTLIST CHANNEL CHAN CDATA #REQUIRED>				//元素CHANNEL的属性“CHAN”，是文本数据
<!ATTLIST PROGRAMSLOT VTR CDATA #IMPLIED>			//元素PROGRAMSLOT的属性“VTR”，是文本数据，可有可无
<!ATTLIST TITLE RATING CDATA #IMPLIED>				//元素TITLE的属性“RATING”，是文本数据，可有可无
<!ATTLIST TITLE LANGUAGE CDATA #IMPLIED>			//元素TITLE的属性“LANGUAGE”，是文本数据，可有可无
